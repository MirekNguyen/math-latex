  \Biggr|
    \begin{alignedat}{2}

    \end{alignedat}
  \Biggr| = \\
  \text{I. způsob} &= \int_{g(a)}^{g(b)}f(t)\,dt = \Bigr[F(t)\Bigr]_{g(a)}^{g(b)} = F(g(b))-F(g(a)) \\
  \text{II. způsob} &= \int_?^?f(t)\,dt=\Bigr[F(t)\Bigr]_?^? = \\
  &=\Bigr[F(t)\Bigr]_a^b = F(b)-F(a) \\
  & \text{zde existuje mez, vrátím substituci}
\end{align*}
