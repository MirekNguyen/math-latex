\subsubsection{Monotonie - příklady}
\begin{align*}
  f(x)&=\frac{x^2}{x+2}
\end{align*}
\hrule
\begin{align*}
  f'(x)&=\frac{x*(x+4)}{(x+2)^2} \\
\end{align*}
