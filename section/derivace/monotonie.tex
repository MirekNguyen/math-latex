\subsubsection{Monotonie - příklady}
\begin{align*}
  f(x)&=\frac{x^2}{x+2} \\
  f'(x)&=\frac{x*(x+4)}{(x+2)^2} \\
\end{align*}
\begin{center}
  rostoucí na $(\infty;-4\rangle \text{ a } (0;\infty)$ \\
  klesající na $\langle-4;-2) \text{ a } (-2;0\rangle$
\end{center}
